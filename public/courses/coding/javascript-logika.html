Pernyataan if...else if...else

codeSintaks Dasar
if (kondisi1) {
  // kode jika kondisi1 benar
} else if (kondisi2) {
  // kode jika kondisi2 benar
} else {
  // kode jika semua salah
}
lightbulbCara Kerja
looks_oneEvaluasi kondisi1
looks_twoJika salah → evaluasi kondisi2
looks_3Jika semua salah → jalankan else
play_circleContoh
let nilai = 75;

if (nilai >= 90) {
  console.log("Sangat Baik");
} else if (nilai >= 70) {
  console.log("Baik");
} else {
  console.log("Perlu Perbaikan");
}
Output:
Baik

Operator Logika

settingsJenis Operator Logika
&&
AND
True jika KEDUA kondisi benar
||
OR
True jika SALAH SATU benar
!
NOT
Membalik nilai boolean
table_chartTabel Kebenaran Singkat
ABA && BA || B
TTTT
TFFT
FFFF
codeContoh Penggunaan
let umur = 25;
let punyaSIM = true;

// Operator AND
if (umur >= 17 && punyaSIM) {
  console.log("Boleh mengemudi");
}

// Operator OR
if (umur < 17 || !punyaSIM) {
  console.log("Dilarang mengemudi");
}
Output:
Boleh mengemudi

}
